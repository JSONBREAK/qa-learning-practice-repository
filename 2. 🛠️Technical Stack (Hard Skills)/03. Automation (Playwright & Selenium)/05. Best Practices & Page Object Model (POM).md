## 3. ‡∏ß‡∏¥‡∏ò‡∏µ‡πÄ‡∏Ç‡∏µ‡∏¢‡∏ô‡πÑ‡∏ü‡∏•‡πå Page (LoginPage.ts)

```
import { Page, Locator } from '@playwright/test';

export class LoginPage {
  readonly page: Page;
  readonly userField: Locator;
  readonly passField: Locator;
  readonly loginBtn: Locator;

  constructor(page: Page) {
    this.page = page;
    this.userField = page.locator('#user');
    this.passField = page.locator('#pass');
    this.loginBtn = page.locator('#login-button');
  }

  async login(user: string, pass: string) {
    await this.userField.fill(user);
    await this.passField.fill(pass);
    await this.loginBtn.click();
  }
}
```
